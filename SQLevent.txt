//CREATE A DATE CREATED COLUMN IN THE DATABASE FOR THE CATERING DATA USING DEFAULT_VALUE=GETDATE()
//THEN ADD EVENT THAT DELETES COLUMNS WITH THE DATE CREATED GREATER THAN 2 YEARS


SET GLOBAL event_scheduler = 1; //makes event scheduler usable.
CREATE EVENT IF NOT EXISTS scheduledDeletion
ON SCHEDULE EVERY '1' MONTH
STARTS '2019-05-16 00:00:00'
DO 
	DELETE FROM cateringdata?????
	WHERE ??DATECOLUMN??< DATE_SUB(NOW(), INTERVAL 2 YEAR);
